import { Meta, ColorPalette, ColorItem } from '@storybook/blocks';
import { light } from '../../../../packages/colour-system/raw/colours.json';

<Meta title="Guides / Colour System / Colors Light" />

# Colors for Light Mode

<ColorPalette>
  {Object.keys(light).map(key => (
    <>
      <ColorItem
        title={`${key}`}
        subtitle={`light.${key}`}
        colors={Object.keys(light[key])
          .slice(0, 6)
          .reduce((acc, subKey) => {
            acc[subKey] = light[key][subKey].value;
            return acc;
          }, {})}
      />
      <ColorItem
        title={`${key}`}
        subtitle={`light.${key}`}
        colors={Object.keys(light[key])
          .slice(6, 12)
          .reduce((acc, subKey) => {
            acc[subKey] = light[key][subKey].value;
            return acc;
          }, {})}
      />
      {Object.keys(light[key]).length > 12 && (
        <ColorItem
          title={`${key}`}
          subtitle={`light.${key}`}
          colors={Object.keys(light[key])
            .slice(12, 18)
            .reduce((acc, subKey) => {
              acc[subKey] = light[key][subKey].value;
              return acc;
            }, {})}
        />
      )}
    </>
  ))}
</ColorPalette>
